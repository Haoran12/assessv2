import { ElMessage } from 'element-plus'
import SidebarMenuItem from './components/SidebarMenuItem.vue'

const route = useRoute()

const appStore = useAppStore()

const currentYear = computed({
  get: () => appStore.currentYearId,
  set: (val) => appStore.setCurrentYear(val)
})
const years = computed(() => appStore.years)
const cachedViews = computed(() => appStore.cachedViews)

// 当前激活的菜单
const activeMenu = computed(() => route.path)

// 面包屑导航
const breadcrumbs = computed(() => {
  const matched = route.matched.filter(item => item.meta?.title)
  return matched.map(item => ({
    path: item.path,
    title: item.meta.title as string
  }))
})

// 菜单路由（根据权限过滤）
const menuRoutes = computed(() => {
  return router.options.routes[1].children?.filter(route => {
    if (!route.meta?.title) return false
    if (route.meta.permission) {
      return userStore.hasPermission(route.meta.permission as string)
    }
    return true
  })
})

// 切换侧边栏
const toggleSidebar = () => {
  appStore.toggleSidebar()
}

// 年度切换
const handleYearChange = (yearId: number) => {
  ElMessage.success(`已切换到 ${years.value.find(y => y.id === yearId)?.name}`)
  // 刷新当前页面数据
  router.go(0)
}

// 用户操作
const handleUserCommand = (command: string) => {
  switch (command) {
    case 'profile':
      // 打开个人信息对话框
      break
    case 'password':
      // 打开修改密码对话框
      break
    case 'logout':
      userStore.logout()
      router.push('/login')
      break
  }
}
</script>

<template>
  <div class="login-container">
    <div class="login-box">

        class="login-form"
        @keyup.enter="handleLogin"
      >
        <el-form-item prop="username">
          <el-input
            v-model="loginForm.username"
            placeholder="请输入用户名"
            size="large"
            :prefix-icon="User"
          />
        </el-form-item>

        <el-form-item prop="password">
          <el-input
            v-model="loginForm.password"
            type="password"
            placeholder="请输入密码"

            :prefix-icon="Lock"
            show-password
          />
        </el-form-item>

        <el-form-item>
          <el-button
            type="primary"
            size="large"
            :loading="loading"
            class="login-button"
            @click="handleLogin"
          >
            登录
          </el-button>
        </el-form-item>
      </el-form>

      <div class="login-footer">
        <span>版本 v1.0.0</span>
      </div>
    </div>
  </div>

<script setup lang="ts">
import { ref, reactive } from 'vue'
import { useRouter, useRoute } from 'vue-router'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'
import { User, Lock } from '@element-plus/icons-vue'
import type { FormInstance, FormRules } from 'element-plus'

const router = useRouter()
const route = useRoute()
const userStore = useUserStore()

const loginFormRef = ref<FormInstance>()
const loading = ref(false)

const loginForm = reactive({
  username: '',
  password: ''
})

const loginRules: FormRules = {
  username: [
    { required: true, message: '请输入用户名', trigger: 'blur' }
  ],
  password: [
    { required: true, message: '请输入密码', trigger: 'blur' },
    { min: 6, message: '密码长度不能少于6位', trigger: 'blur' }
  ]
}

const handleLogin = async () => {
  if (!loginFormRef.value) return

  await loginFormRef.value.validate(async (valid) => {
    if (!valid) return

    loading.value = true
    try {
      await userStore.login(loginForm.username, loginForm.password)
      ElMessage.success('登录成功')

      const redirect = route.query.redirect as string || '/dashboard'
      router.push(redirect)
    } catch (error: any) {
      ElMessage.error(error.message || '登录失败')
    } finally {
      loading.value = false
    }
  })
}
</script>

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { login as apiLogin, getUserInfo, logout as apiLogout } from '@/api/auth'
import type { UserInfo, LoginParams } from '@/types/user'

export const useUserStore = defineStore('user', () => {
  const token = ref<string>('')
  const userInfo = ref<UserInfo | null>(null)
  const permissions = ref<string[]>([])

  const isLoggedIn = computed(() => !!token.value)
  const username = computed(() => userInfo.value?.username || '')
  const avatar = computed(() => userInfo.value?.avatar || '')
  const role = computed(() => userInfo.value?.role || '')

  // 登录
  const login = async (username: string, password: string) => {
    const res = await apiLogin({ username, password })
    token.value = res.token
    localStorage.setItem('token', res.token)

    await fetchUserInfo()
  }

  // 获取用户信息
  const fetchUserInfo = async () => {
    const res = await getUserInfo()
    userInfo.value = res.user
    permissions.value = res.permissions
  }

  // 登出
  const logout = async () => {
    await apiLogout()
    token.value = ''
    userInfo.value = null
    permissions.value = []
    localStorage.removeItem('token')
  }

  // 检查权限
  const hasPermission = (permission: string): boolean => {
    return permissions.value.includes(permission) || permissions.value.includes('*')
  }

  // 检查角色
  const hasRole = (role: string): boolean => {
    return userInfo.value?.role === role
  }

  // 初始化（从 localStorage 恢复）
  const init = async () => {
    const savedToken = localStorage.getItem('token')
    if (savedToken) {
      token.value = savedToken
      try {
        await fetchUserInfo()
      } catch (error) {
        // Token 失效，清除
        logout()
      }
    }
  }

  return {
    token,
    userInfo,
    permissions,
    isLoggedIn,
    username,
    avatar,
    role,
    login,
    logout,
    fetchUserInfo,
    hasPermission,
    hasRole,
    init
  }
})

// src/stores/app.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getYears, getCurrentYear } from '@/api/assessment'
import type { AssessmentYear } from '@/types/assessment'

export const useAppStore = defineStore('app', () => {
  const sidebarCollapsed = ref(false)
  const years = ref<AssessmentYear[]>([])
  const currentYearId = ref<number | null>(null)
  const cachedViews = ref<string[]>([])

  const currentYear = computed(() => {
    return years.value.find(y => y.id === currentYearId.value)
  })

  // 切换侧边栏
  const toggleSidebar = () => {
    sidebarCollapsed.value = !sidebarCollapsed.value
    localStorage.setItem('sidebarCollapsed', String(sidebarCollapsed.value))
  }

  // 加载年度列表
  const loadYears = async () => {
    const res = await getYears()
    years.value = res.data

    // 设置当前年度
    if (!currentYearId.value && years.value.length > 0) {
      const current = await getCurrentYear()
      currentYearId.value = current.id
    }
  }

  // 设置当前年度
  const setCurrentYear = (yearId: number) => {
    currentYearId.value = yearId
    localStorage.setItem('currentYearId', String(yearId))
  }

  // 添加缓存视图
  const addCachedView = (name: string) => {
    if (!cachedViews.value.includes(name)) {
      cachedViews.value.push(name)
    }
  }

  // 移除缓存视图
  const removeCachedView = (name: string) => {
    const index = cachedViews.value.indexOf(name)
    if (index > -1) {
      cachedViews.value.splice(index, 1)
    }
  }

  // 初始化
  const init = async () => {
    // 恢复侧边栏状态
    const savedCollapsed = localStorage.getItem('sidebarCollapsed')
    if (savedCollapsed) {
      sidebarCollapsed.value = savedCollapsed === 'true'
    }

    // 恢复当前年度
    const savedYearId = localStorage.getItem('currentYearId')
    if (savedYearId) {
      currentYearId.value = Number(savedYearId)
    }

    // 加载年度列表
    await loadYears()
  }

  return {
    sidebarCollapsed,
    years,
    currentYearId,
    currentYear,
    cachedViews,
    toggleSidebar,
    loadYears,
    setCurrentYear,
    addCachedView,
    removeCachedView,
    init
  }
})

    return cachedData.value.get(key)

  return fetcher().then(data => {
    cachedData.value.set(key, data)
    return data
  })
}
```
